% ── PORTADA ─────────────────────────────────────────────────
\makeatletter
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \coverFont
    \newgeometry{left=6cm, top=2cm, bottom=2cm, right=2cm}
    \newpagecolor{titlepage-bg}\afterpage{\restorepagecolor}
    \thispagestyle{empty}

    \begin{tikzpicture}[remember picture, overlay]
      \fill[sidebar-cover]
        (current page.north west)
        rectangle ([xshift=4.7cm]current page.south west);
      \draw[sidebar-cover, line width=2.5pt, opacity=0.85]
        ([xshift=4.7cm]current page.north west)
        -- ([xshift=4.7cm]current page.south west);
    \end{tikzpicture}

    \begin{flushleft}
    \noindent\\[-1em]
    \color{title-text}
    \makebox[0pt][l]{\textcolor{title-text}{\rule{1.3\textwidth}{3pt}}}
    \par\noindent
    {
      \setstretch{1.4}
      \vfill

      {\Huge\bfseries\color{title-text} $title$}

      $if(subtitle)$
      \vskip 0.8em
      {\Large\color{subtitle-text} $subtitle$}
      $endif$

      \vskip 3.5em

      $for(author)$
      {\large\color{subtitle-text} $author$}\\[0.4em]
      $endfor$

      $if(version)$
      \vskip 0.4em
      {\normalsize\bfseries\color{title-text} $version$}
      $endif$

      $if(description)$
      \vskip 1.8em
      {\small\color{subtitle-text} $description$}
      $endif$

      \vfill
    }
    {\normalsize\color{subtitle-text} $date$}
    \vskip 1em
    \end{flushleft}
  \end{titlepage}
  \restoregeometry
  \setcounter{page}{1}
}
\makeatother
